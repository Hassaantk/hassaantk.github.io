Here's a script you can use: ```javascript function generateEmails(numEmails) { var emails = []; var characters = "abcdefghijklmnopqrstuvwxyz0123456789"; for (var i = 0; i < numEmails; i++) { var password = ''; var email = ''; // Generate random password for (var j = 0; j < 8; j++) { var randomIndex = Math.floor(Math.random() * characters.length); password += characters.charAt(randomIndex); } // Generate email address email = password + "@example.com"; emails.push({ email: email, password: password }); } return emails; } function displayEmails() { var numEmails = 10; // Change this value to generate a different number of emails var emails = generateEmails(numEmails); // Display the emails and passwords for (var i = 0; i < emails.length; i++) { console.log("Email: " + emails[i].email + " | Password: " + emails[i].password); } } // Call the displayEmails function when the button is clicked document.getElementById('generateButton').addEventListener('click', displayEmails); ``` To use this script, include it in your HTML file where you want the button to appear.
